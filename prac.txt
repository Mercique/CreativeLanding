function cardSwap(value) {
  var yearly = document.getElementById("yearly");
  var monthly = document.getElementById("monthly");
  if (value == 'Yearly') {
    yearly.className = 'db';
  } else {
    yearly.className = 'dn';
  }
  if (value == 'Monthly') {
    monthly.className = 'db';
  } else {
    monthly.className = 'dn';
  }
}
